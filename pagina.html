                    const { error: upErr } = await _supabase.storage.from('avatares.').upload(filePath, file);

                    if (upErr) throw upErr;

                    

                    const { data } = _supabase.storage.from('avatares.').getPublicUrl(filePath);

            } catch (err) { alert("Error: " + err.message); }
